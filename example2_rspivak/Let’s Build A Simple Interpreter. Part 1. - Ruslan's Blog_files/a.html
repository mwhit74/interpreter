                                def main():
                                    while True:
                                        try:
                                            # To run under both python 2 & Python3                        
                                            if 3 == sys.version_info.major:
                                                text = input('calc> ')
                                            if 2 == sys.version_info.major:
                                                text = raw_input('calc> ')
                                        except EOFError:
                                            break
                                        if not text:
                                            continue
                                        interpreter = Interpreter(text)
                                        result = interpreter.expr()
                                        print(result)
